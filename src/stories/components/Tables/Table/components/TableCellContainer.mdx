<Title>TableCellContainer Component</Title>

The `TableCellContainer` component is used to wrap the body cells of the table, providing a container for each row of data. It includes hover and checked state management, and optional sub-table functionality.
## Component Styles & Features

### Default Style 
<TableCellContainer isChecked={false} >
      <TableCell cellLabel="Row 1, Cell 1" />
      <TableCell cellLabel="Row 1, Cell 2" />
</TableCellContainer>

<Source
  code={`
import React, { useState } from 'react';
import { TableCellContainer, TableCell } from 'torch-glare';

const ExampleTableCellContainer = () => {
  const [isChecked, setIsChecked] = useState(false);

  const handleCheckBoxChange = () => {
    setIsChecked(!isChecked);
  };

  return (
    <table>
      <tbody>
        <TableCellContainer isChecked={isChecked} onCheckBoxChange={handleCheckBoxChange}>
          <TableCell cellLabel="Row 1, Cell 1" />
          <TableCell cellLabel="Row 1, Cell 2" />
        </TableCellContainer>
      </tbody>
    </table>
  );
};

export default ExampleTableCellContainer;
  `}
