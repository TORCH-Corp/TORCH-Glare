## Usage

```jsx
import { NoteInputField } from 'your-component-library';

const App = () => (
  <div>
    <NoteInputField
        name="note"
        label="Your Note"
        placeholder="Enter your note here..."
        required_label="*"
        secondary_label="All fields are mandatory"
        label_style="vertical"
    />
  </div>
);

export default App;
```
