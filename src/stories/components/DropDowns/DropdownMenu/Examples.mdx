## Usage

```jsx
import { DropDownMenu, DropDownMenuItem } from 'torch-glare';

const App = () => (
  <div>
    <DropDownMenu component_style="System-Style">
      <DropDownMenuItem component_style='System-Style' component_label={'Option Example'} element_name={'example'} />
      <DropDownMenuItem component_style='System-Style' component_label={'Option Another'} element_name={'another'} />
    </DropDownMenu>
  </div>
);

export default App;
```
