## Usage

```jsx
import { DropDownButtonResultShow, DropDownMenu, DropDownMenuItem } from 'torch-glare';

const App = () => (
  <div>
    <DropDownButtonResultShow
      component_size="M"
      selected_value="Example Dropdown"
      drop_down_list_child={
        <DropDownMenu>
          <DropDownMenuItem component_style='System-Style' component_label={'Option Example'} element_name={'example'} />
        </DropDownMenu>
      }
    />
  </div>
);

export default App;
```
