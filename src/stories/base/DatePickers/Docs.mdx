## Usage

```jsx
import { Datepicker } from 'torch-glare';

const App = () => (
  <div>
    <Datepicker
      component_style='system'
      component_size="M"
      placeholder="Select date"
      name="example-datepicker"
    />
  </div>
);

export default App;
