## Usage

```jsx
import { DropDownMenuItem } from 'torch-glare';

const App = () => (
 <DropDownMenu  component_style='System-Style'>
    <DropDownMenuItem
      component_label="example 1"
      element_name="example1"
      component_style="System-Style"
    />
    <DropDownMenuItem
      component_label="example 2"
      element_name="example2"
      component_style="System-Style"
    />
        <DropDownMenuItem
      component_label="example 3"
      element_name="example3"
      component_style="System-Style"
    />
    <DropDownMenuItem
      component_label="example 4"
      element_name="example4"
      component_style="System-Style"
    />
  </DropDownMenu>

);

export default App;
```
