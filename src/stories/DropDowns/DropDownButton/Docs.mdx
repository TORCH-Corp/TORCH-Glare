## Usage

```jsx
import { DropDownButton, DropDownMenu, DropDownMenuItem } from 'torch-glare';

const App = () => (
  <div>
    <DropDownButton
      component_size="M"
      component_label="Example Dropdown"
      drop_down_list_child={
        <DropDownMenu>
          <DropDownMenuItem component_style='System-Style' component_label={'Option Example'} element_name={'example'} />
        </DropDownMenu>
      }
    />
  </div>
);

export default App;
```
